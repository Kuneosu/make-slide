// Configuratio principalis
const config = {
  theme: 'gradient-pop',
  transitions: {
    duration: 600,
    easing: 'cubic-bezier(0.25, 0.1, 0.25, 1)',
    type: 'fade'
  },
  features: {
    progressBar: true,
    slideCounter: true,
    speakerNotes: true,
    touchNavigation: true
  },
  gradient: {
    start: '#6366f1',
    mid: '#ec4899',
    end: '#f97316'
  }
};

function initSlides(options) {
  const deck = document.getElementById('deck');
  const slides = deck.querySelectorAll('.slide');

  slides.forEach((slide, index) => {
    slide.dataset.index = index;
    applyTransition(slide, options.transitions);
  });

  return { deck, slides, total: slides.length };
}
